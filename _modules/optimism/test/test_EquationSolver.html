def energy(x, params):
    p = params[0]
    return x[0]*(x[0]+1) + 0.3*x[1]*(x[1]-0.2) + 0.2*x[2]*(x[2]-0.5) + x[0]*x[0]*x[1]*x[1] + p*x[0]*x[1] + np.sin(x[0])
